# ansible-core 2.16 is the last release that supports managing legacy hosts
# (Ubuntu 18.04, CentOS 8 — Python 3.6); its managed-node minimum is Python 3.6.
# 2.17+ requires Python 3.7+ on the managed node, and its apt/dnf modules fail on
# legacy hosts because the python-apt/dnf bindings force a respawn under python3.6.
ansible-core>=2.16,<2.17

# The password_hash('sha512') filter runs on the controller. On macOS, crypt.crypt
# does not support sha512, so passlib is required (used by the `users` role to
# hash passwords).
passlib>=1.7
